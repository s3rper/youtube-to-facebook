<script>
document.addEventListener("DOMContentLoaded", async () => {
  const apiUrl = "https://api.seeip.org/geoip";

  // Retrieve a cookie by name
  function getCookie(name) {
    const value = `; ${document.cookie}`;
    const parts = value.split(`; ${name}=`);
    if (parts.length === 2) return parts.pop().split(";").shift();
  }

  // Set a cookie
  function setCookie(name, value, days) {
    const expires = new Date();
    expires.setTime(expires.getTime() + days * 24 * 60 * 60 * 1000);
    document.cookie = `${name}=${value};expires=${expires.toUTCString()};path=/`;
  }

  function getLanguage() {
    return window.location.pathname.includes("/es/") ? "es" : "en";
  }

  // Function to fetch user region from API
  async function fetchUserRegion() {
    try {
      const response = await fetch(apiUrl);
      if (!response.ok) throw new Error("Failed to fetch region");

      const data = await response.json();
      console.log("GeoIP Data:", data);

      return data.country_code ? data.country_code.toLowerCase() : "us"; // Default to US
    } catch (error) {
      console.error("Error fetching GeoIP:", error);
      return "us"; // Fallback in case of error
    }
  }

  // Reset rhRegion if missing or incorrect, fetch region dynamically
  async function resetRhRegion() {
    let region = "none"; // Set to none initially
    setCookie("rhRegion", region, 7);

    const detectedRegion = await fetchUserRegion();

    const regionMapping = {
      us: "us",
      ca: "ca",
      gb: "uk",
      au: "au",
      nz: "nz",
      br: "br",
      join: "join",
      mx: "latam", // Latin America ES
      ar: "latam",
      co: "latam",
      pe: "latam",
      cl: "latam",
      ec: "latam",
      uy: "latam",
      bo: "latam",
      py: "latam",
      ve: "latam",
      cr: "latam",
      do: "latam",
      pa: "latam",
      gt: "latam",
      hn: "latam",
      sv: "latam",
      ni: "latam",
      pr: "latam",
      sa: "sa",
    };

    region = regionMapping[detectedRegion] || "us"; // Default to US if not listed

    setCookie("rhRegion", region, 7);
    return region;
  }

  const tourMappings = {
    "/product/chair-yoga": "/tour/yoga-28-day-challenge-generic-us",
    "/es/product/yoga-silla": "/tour/yoga-28-day-challenge-generic-us",
    "/product/weight-loss": "/tour/nutrition-monthly-generic-us",
    "/es/product/perdida-peso": "/tour/nutrition-monthly-generic-us",
    "/product/keto": "/tour/nutrition-keto-weekly",
    "/es/product/dieta-keto-cetogenica": "/tour/nutrition-keto-weekly",
    "/product/wall-pilates": "/tour/wall-pilates-challenge-us",
    "/es/product/pilates-pared": "/tour/wall-pilates-challenge-us",
  };

  function updateHeaderQuizCTAByUrl() {
    const currentUrl = window.location.href;
    const url = new URL(currentUrl);

    for (const [productPath, tourPath] of Object.entries(tourMappings)) {
      if (currentUrl.includes(productPath)) {
        url.pathname = tourPath;
        updateHeaderQuizCTA(url.toString());
        return;
      }
    }
  }

  function updateHeaderQuizCTA(url) {
    const linkElement = document.getElementById("header-quiz-cta");
    if (linkElement && linkElement.tagName === "A") {
      linkElement.setAttribute("href", url);
    }
  }

  // Update anchor URLs with language and region parameters
  async function updateAnchors() {
    const region = await resetRhRegion(); // Ensure rhRegion is correctly set
    const language = getLanguage();

    document.querySelectorAll("a").forEach((anchor) => {
      try {
        const url = new URL(anchor.href);
        const params = new URLSearchParams(url.search);

        for (const [_, tourPath] of Object.entries(tourMappings)) {
          if (url.pathname.startsWith(tourPath)) {
            params.set("lang", language);
            params.set("region", region);

            // Adjust path for specific regions
            if (region === "latam") {
              url.pathname = tourPath.replace("-us", "-latam");
            } else if (region === "br") {
              url.pathname = tourPath.replace("-us", "-br");
              params.set("lang", "pt");
            } else if (region !== "us") {
              url.pathname = tourPath.replace("-us", "-others");
            }

            url.search = params.toString();
            anchor.href = url.toString();
          }
        }
      } catch (error) {
        console.error(error);
      }
    });
  }

  const sendGTMEvent = () => {
    if (window.dataLayer) {
      window.dataLayer.push({ event: "homepage_loaded" });
    }
  };

  // Initialize script
  updateHeaderQuizCTAByUrl();
  await updateAnchors();
  sendGTMEvent();
});
</script>
